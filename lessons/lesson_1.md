# Урок 1 - Имена переменных

webdriver - webdriver_path // переменная хранила именно путь к webdriver, а не класс

db - database_session // переменная относилась к сессии

now - datetime_now // неочевидное односложное название переменной

need_to_start - target_actual_diff // переменная дожна была отражать количество инстанций, которые нужно запустить, чтобы достичь target

js_code - get_elements_js // выполняемый код записан в переменную, название не отражает назначение

mail - imap_client // более ясное название переменной, хранящей соединение по протоколу IMAP

cats - cat_instances_list // более подробное описание, что в переменной хранится список экземпляров класса

len1 - list1_len // указание что длина относится именно к списку

resp - response // сокращение полного слова

size - block_size // более подробное описание, к чему относится

sl - string_list // строка, переведенная в список

me_url - auth_token_url // более четкое название (до этого была так названа потому что содержала /me в конце url)
